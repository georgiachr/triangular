//returns a video list based on limit and skip values
//or error
module.exports.getVideoList = function(req) {

    var limit = req.param("limit");
    var skip = req.param("skip") - 1;
    var videoList = [];

    Video.getVideoListByPage(limit,skip)
        .then(function(videos) {

            _.each(videos, function (video, index) {

                videoList.push({
                    id: video.id,
                    sname: video.sname,
                    eventdatetype: video.eventdatetype
                });

            });

            console.log('videolist in VideoReqProcessing = '+JSON.stringify(videoList));

            return (videoList);
        })
        .fail(function(err){

            throw err;

       });

};

/**
 * Supposed that token is authorized (isTokenAuthorized).
 * Search for user's email. If email found then compare tokens.
 * @param req
 * @param res
 * @param next
 */
module.exports = function (req, res, next) {

    var header_token;
    var useridFromHeader;
    var decoded;

    if (req.headers) {

        if (req.headers["x-auth-token"] === undefined) {
            return res.json(401, {err: 'isUserTokenValid POLICY: No Authorization header was found'});
        }
        else {
            //taken from isTokenAuthorized
            header_token = req.headers["x-auth-token"];

            decoded = jwtToken.decodeToken(header_token, {complete: true});

            useridFromHeader = decoded.payload.userid;

            //Waterline's query methods using callbacks.
            //Promises are supported by default
            User.findOne({
                id: useridFromHeader
            }, function foundUser(err, user) {

                if (err) {
                    console.log("isUserTokenValid POLICY: Found user error!");
                    return res.negotiate(err);
                }

                if (!user) {
                    console.log("isUserTokenValid POLICY: No user found!");
                    return res.notFound();
                }
                else {
                    if (user.token != header_token) {
                        console.log("isUserTokenValid POLICY: User's token is not valid or it has expired!");
                        return res.json(401, {err: 'isUserTokenValid POLICY: The token is not valid'});
                    }
                    else{
                        console.log("isUserTokenValid POLICY: User's token is valid!");
                        next();
                    }

                }

            });

        }

    } // if (req.headers)

};

    var adapterType = Video.connections[

        ].config.adapter;

    if(adapterType == 'sails-mysql'){

    var limit = req.param("limit");
    var skip = req.param("skip") - 1;

    var command =

        "FROM videoreport " +
        "GROUP BY sname " +
        "ORDER BY views " + 'DESC' +

        " OFFSET " + skip;
